- 组件基础引用

  ```html
  <template>
    <el-table :data="tableFilterData()" style="width: 100%">
      <el-table-column prop="date" label="Date" width="180" />
      <el-table-column prop="name" label="Name" width="180" />
      <el-table-column prop="address" label="Address" />
    </el-table>
    <div class="page-normal">
      <el-pagination
        background
        :small="true"
        v-model:current-page="page"
        v-model:page-size="limit"
        layout="total,prev,pager,next,sizes,jumper"
        :total="total"
        :page-sizes="[10, 20, 30, 40, 50, 100]"
        @current-change="handleCurrentChange"
        @size-change="handleSizeChange"
      />
    </div>
  </template>
  ```

- 定义需要用到的变量

  ```js
  import { ref, reactive, toRefs } from "vue";
  const ruleForm =
    reactive <
    any >
    {
      tableList: [
        {
          date: "2016-05-03",
          name: "Tom",
          address: "No. 189, Grove St, Los Angeles",
        },
        {
          date: "2016-05-02",
          name: "Tom",
          address: "No. 189, Grove St, Los Angeles",
        },
        {
          date: "2016-05-04",
          name: "Tom",
          address: "No. 189, Grove St, Los Angeles",
        },
        {
          date: "2016-05-01",
          name: "Tom",
          address: "No. 189, Grove St, Los Angeles",
        },
      ],
      rules: {
        paramName: [{ validator: paramNamePass, trigger: "change" }],
        opType: [
          { required: true, message: "操作符不能为空", trigger: "change" },
        ],
      },
    };
  const state = reactive({
    page: 1,
    limit: 20,
    total: 0,
  });

  const { tableList, rules } = toRefs(ruleForm);
  const { total, limit, page } = toRefs(state);

  /**
   * @description 前端限制分页（tableData为当前展示页表格）
   */
  const tableFilterData = () => {
    let res = tableList.value;
    state.total = res.length;
    return res.filter(
      (_item, index) =>
        index < page.value * limit.value &&
        index >= limit.value * (page.value - 1)
    );
  };

  /**
   * @description 改变页码
   * @param e
   */
  const handleCurrentChange = (e) => {
    state.page = e;
  };

  /**
   * @description 改变页数限制
   * @param e
   */
  const handleSizeChange = (e) => {
    state.limit = e;
  };
  ```
